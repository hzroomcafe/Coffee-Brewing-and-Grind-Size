☕ Apps Script 程式碼 — 直接從這裡複製

點下方按鈕複製全部程式碼，再貼到 Apps Script 取代原有內容。這個頁面不會有 HTML 實體問題。

// ===== 只改這三行 =====
const SHEET_ID   = '1XachRqnXwZ3oaCHv27vjLZ3uwkBKxMMdkDz3ySElhnM';
const SHEET_NAME = '投稿';
const NOTIFY_TO  = '[email protected]';

const REPORT_SHEET = '回報';
const MAX_PER_FP_HOUR = 5;

function doPost(e) {
  try {
    const d = JSON.parse(e.postData.contents);

    // ── 回報錯誤 ──
    if (d._action === 'report') {
      const ss = SpreadsheetApp.openById(SHEET_ID);
      let rsh = ss.getSheetByName(REPORT_SHEET);
      if (!rsh) {
        rsh = ss.insertSheet(REPORT_SHEET);
        rsh.appendRow(['時間','店名','錯誤類型','說明','狀態']);
        rsh.getRange(1,1,1,5).setFontWeight('bold').setBackground('#f0ece4');
      }
      rsh.appendRow([new Date(), d.cafeName||'', d.reportType||'', d.desc||'', '待處理']);
      rsh.getRange(rsh.getLastRow(), 1, 1, 5).setBackground('#FFF3CD');
      MailApp.sendEmail(NOTIFY_TO,
        '⚠️ 咖啡廳資料回報：' + (d.cafeName||'未知'),
        '店家：' + (d.cafeName||'') + '\n類型：' + (d.reportType||'') + '\n說明：' + (d.desc||'') +
        '\n\n請到試算表「回報」分頁（黃色列）確認並處理。');
      return ok();
    }

    // ── 評論 ──
    if (d._action === 'review') {
      const ss = SpreadsheetApp.openById(SHEET_ID);
      let vsh = ss.getSheetByName('評論');
      if (!vsh) {
        vsh = ss.insertSheet('評論');
        vsh.appendRow(['時間','店名','名稱','評論']);
        vsh.getRange(1,1,1,4).setFontWeight('bold').setBackground('#f0ece4');
      }
      const by = String(d.by||'').slice(0,20);
      const text = String(d.text||'').slice(0,100);
      if (!d.cafeName || !by || !text) return ok();
      vsh.appendRow([new Date(), d.cafeName, by, text]);
      return ok();
    }

    // ── 新投稿 ──
    if (d.website) return ok();
    if (!d.name || !d.by || !d.reason || !d.tags) return ok();
    const reason = String(d.reason);
    if (reason.length < 5 || reason.length > 300) return ok();

    const props = PropertiesService.getScriptProperties();
    const key = 'rl_' + (d._fp || 'anon');
    const now = Date.now();
    const hist = JSON.parse(props.getProperty(key) || '[]').filter(t => now - t < 3600000);
    if (hist.length >= MAX_PER_FP_HOUR) return ok();
    hist.push(now); props.setProperty(key, JSON.stringify(hist));

    const sh = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEET_NAME);
    const recent = sh.getDataRange().getValues().slice(-60);
    const dup = recent.some(r => r[1] === d.name && r[6] === d.by &&
                (now - new Date(r[0]).getTime()) < 86400000);
    if (dup) return ok();

    sh.appendRow([
      new Date(), d.name, d.area||'', d.address||'',
      d.lat||'', d.lng||'', d.by, reason,
      d.tags, d.link||'', '待審核', d.photos||''
    ]);

    MailApp.sendEmail(NOTIFY_TO, '☕ 新咖啡廳推薦待審核',
      '店名：' + d.name + '\n推薦人：' + d.by + '\n原因：' + reason +
      '\n標籤：' + d.tags + '\n地址：' + (d.address||'(無)') +
      (d.photos && d.photos !== '[]' ? '\n（附有照片）' : '') +
      '\n\n到試算表把「狀態」改成「已通過」即可上架。');
    return ok();
  } catch (err) { return ok(); }
}

function ok() {
  return ContentService.createTextOutput('{"ok":true}')
    .setMimeType(ContentService.MimeType.JSON);
}
部署步驟：
1. 點上方「複製全部程式碼」
2. 打開 Apps Script（試算表 → 擴充功能 → Apps Script）
3. 全選（Ctrl+A / ⌘A）→ 刪除 → 貼上（Ctrl+V / ⌘V）
4. 按 💾 儲存
5. 部署 → 管理部署作業 → 鉛筆圖示 → 版本選「建立新版本」→ 部署
6. 這步是關鍵！不建立新版本的話改動不會生效
